# list the devices connected to the pc and their status
.\adb.exe devices

# pull a package from the device
.\adb.exe pull /.../application.apk .\...\backup.apk

# start an interactive shell
.\adb.exe shell

# list the packages on the device in lexicographical order
.\adb.exe shell 'pm list packages -f' | sort > ..\default.txt

# remove package from the device without rooting it
.\adb.exe shell 'pm uninstall -k --user 0 com.huawei.android.totemweather'
